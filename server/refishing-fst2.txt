# German
def PGC "b " | "d " | "f " | "g " | "gʷ " | "j " | "k " | "kʷ " | "kː "
| "l " | "lː " | "m " | "mː " | "n " | "nː " | "p " | "pː " | "r " | "s
" | "t " | "w " | "x " | "xʷ " | "z " | "θ ";

def PGV "a " | "ai " | "au " | "e " | "eu " | "eː " | "i " | "iː " | "o
" | "oː " | "u " | "uː ";

def PGSyllable PGC* PGV* ;

def PGWord PGSyllable* ;

def Grimm "p " -> "pf ", "t " -> "ts ";

def OE "oː " -> "ə ";

def SK "s k " -> "ʃ ";

regex PGWord ;
regex Grimm .o. OE .o. SK;

def German PGWord .o. Grimm;

clear
push German
save stack german.bin
